CLASS Field

Attributes
    PRIVATE fieldId : Number
    PRIVATE fieldName : TEXT
    PRIVATE crop : Crop
    PRIVATE activities : LIST<Activity>


CONSTRUCTOR(n : TEXT, c : Crop)

    SET fieldName TO n
    SET crop TO c
    SET activities TO EMPTY LIST

END CONSTRUCTOR



METHODS


    addActivity(newActivity : Activity)

        IF newActivity.type = "Watering"

            IF newActivity.amount < crop.waterNeed

                DISPLAY "Insufficient Water"

            ELSE

                ADD newActivity TO activities

            END IF

        ELSE

            ADD newActivity TO activities

        END IF


    END METHOD



    calculateEstimatedYield()

        totalActivities = LENGTH(activities)

        IF totalActivities = 0

            RETURN 0

        ELSE

            properCare = COUNT OF activities WHERE status = "Completed"

            carePercentage = properCare / totalActivities

            estimatedYield = crop.expectedOutput * carePercentage

            RETURN estimatedYield

        END IF

    END METHOD


END CLASS

    SET expectedOutput TO e

END CONSTRUCTOR


METHODS


    getWaterNeed()

        RETURN waterNeed

    END METHOD



    getExpectedOutput()

        RETURN expectedOutput

    END METHOD


END CLASS





CLASS Activity

Attributes
    PRIVATE activityId : Number
    PRIVATE activityType : TEXT
    PRIVATE activityDate : DATE
    PRIVATE amount : REAL
    PRIVATE status : TEXT
    PRIVATE worker : Worker


CONSTRUCTOR(t : TEXT, d : DATE, a : REAL, w : Worker)

    SET activityType TO t
    SET activityDate TO d
    SET amount TO a
    SET worker TO w
    SET status TO "Completed"

END CONSTRUCTOR



METHODS


    getActivityType()

        RETURN activityType

    END METHOD



    getStatus()

        RETURN status

    END METHOD


END CLASS





CLASS Worker

Attributes
    PRIVATE workerId : Number
    PRIVATE workerName : TEXT
    PRIVATE dailyLimit : Number
    PRIVATE assignedActivities : LIST<Activity>


CONSTRUCTOR(n : TEXT, limit : Number)

    SET workerName TO n
    SET dailyLimit TO limit
    SET assignedActivities TO EMPTY LIST

END CONSTRUCTOR



METHODS


    canAcceptActivity()

        RETURN LENGTH(assignedActivities) < dailyLimit

    END METHOD



    assignActivity(newActivity : Activity)

        IF canAcceptActivity()

            ADD newActivity TO assignedActivities

        ELSE

            DISPLAY "Worker daily limit exceeded"

        END IF

    END METHOD


END CLASS
