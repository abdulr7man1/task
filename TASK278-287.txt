    SET backlog TO b

    addTask(newTask : Task)
        IF newTask.developer.canAcceptTask()
            ADD newTask TO tasks
            CALL newTask.developer.assignTask()
        ELSE
            DISPLAY "Developer capacity exceeded"
        END IF
    END METHOD

    calculateProgress()
        completed = COUNT OF tasks WHERE status = "Completed"
        total = LENGTH(tasks)

        IF total = 0
            DISPLAY 0

            DISPLAY (completed * 100) / total
        END IF
    END METHOD

    END METHOD

CLASS Task
Attributes
    taskId : Number
    taskName : TEXT
    status : TEXT
    developer : Developer

CONSTRUCTOR(n : TEXT, s : TEXT, d : Developer)
    SET taskName TO n
    SET status TO s
    SET developer TO d
END CONSTRUCTOR

METHODS
    updateStatus(newStatus : TEXT)
        IF status = "Backlog" AND newStatus = "In Progress"
            SET status TO newStatus
        ELSE IF status = "In Progress" AND newStatus = "Testing"
            SET status TO newStatus
        ELSE IF status = "Testing" AND newStatus = "Completed"
            SET status TO newStatus
        ELSE
            DISPLAY "Invalid Status Transition"
        END IF
    END METHOD

    markComplete()
        IF status = "Testing"
            SET status TO "Completed"
        ELSE
            DISPLAY "Task cannot be completed yet"
        END IF
    END METHOD
