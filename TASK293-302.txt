CLASS Circle EXTENDS Shape

Attributes
    PRIVATE radius : REAL

CONSTRUCTOR(c : TEXT, x : REAL, y : REAL, r : REAL)
    CALL SUPER(c, x, y)
    SET radius TO r
END CONSTRUCTOR

METHODS

    getArea()
        RETURN 3.14159 * radius * radius
    END METHOD

    getPerimeter()
        RETURN 2 * 3.14159 * radius
    END METHOD

END CLASS



CLASS Rectangle EXTENDS Shape

Attributes
    PRIVATE width : REAL
    PRIVATE height : REAL

CONSTRUCTOR(c : TEXT, x : REAL, y : REAL, w : REAL, h : REAL)
    CALL SUPER(c, x, y)
    SET width TO w
    SET height TO h
END CONSTRUCTOR

METHODS

    getArea()
        RETURN width * height
    END METHOD

    getPerimeter()
        RETURN 2 * (width + height)
    END METHOD

END CLASS



CLASS Triangle EXTENDS Shape

Attributes
    PRIVATE side1 : REAL
    PRIVATE side2 : REAL
    PRIVATE side3 : REAL

CONSTRUCTOR(c : TEXT, x : REAL, y : REAL, s1 : REAL, s2 : REAL, s3 : REAL)
    CALL SUPER(c, x, y)
    SET side1 TO s1
    SET side2 TO s2
    SET side3 TO s3
END CONSTRUCTOR

METHODS

    getArea()
        SET semiPerimeter TO (side1 + side2 + side3) / 2
        RETURN SQRT(semiPerimeter * (semiPerimeter - side1) * (semiPerimeter - side2) * (semiPerimeter - side3))
    END METHOD

    getPerimeter()
        RETURN side1 + side2 + side3
    END METHOD

END CLASS

CLASS DebitCard EXTENDS Card

Attributes
    PRIVATE accountBalance : REAL

CONSTRUCTOR(c : TEXT, e : DATE, s : TEXT, ch : Chip, b : REAL)
    CALL SUPER(c, e, s, ch)
    SET accountBalance TO b
END CONSTRUCTOR

METHODS

    authorisePurchase(amount : REAL)
        IF accountBalance >= amount THEN
            SET accountBalance TO accountBalance - amount
            RETURN TRUE
        ELSE
            RETURN FALSE
        END IF
    END METHOD

END CLASS



CLASS CreditCard EXTENDS Card

Attributes
    PRIVATE creditLimit : REAL
    PRIVATE unpaidPurchases : LIST<REAL>

CONSTRUCTOR(c : TEXT, e : DATE, s : TEXT, ch : Chip, l : REAL)
    CALL SUPER(c, e, s, ch)
    SET creditLimit TO l
    SET unpaidPurchases TO EMPTY LIST
END CONSTRUCTOR

METHODS

    calculateOutstandingBalance()
        SET total TO 0

        FOR EACH purchase IN unpaidPurchases
            SET total TO total + purchase
        END FOR

        RETURN total
    END METHOD

    authorisePurchase(amount : REAL)

        IF calculateOutstandingBalance() + amount <= creditLimit THEN
            ADD amount TO unpaidPurchases
            RETURN TRUE
        ELSE
            RETURN FALSE
        END IF

    END METHOD

END CLASS
