    END METHOD

CLASS VehicleClaim EXTENDS Claim

Attributes

    PRIVATE repairEstimate : REAL
    PRIVATE excessAmount : REAL


CONSTRUCTOR(number : TEXT, date : DATE, folder : DocumentationFolder, repair : REAL, excess : REAL)

    CALL SUPER(number, date, folder)

    SET repairEstimate TO repair
    SET excessAmount TO excess

END CONSTRUCTOR


METHODS

    calculatePayout()

        SET payout TO repairEstimate - excessAmount

        IF payout < 0 THEN
            RETURN 0
        ELSE
            RETURN payout
        END IF

    END METHOD


END CLASS



CLASS MedicalClaim EXTENDS Claim

Attributes

    PRIVATE treatmentCost : REAL
    PRIVATE medicalCoverageLimit : REAL


CONSTRUCTOR(number : TEXT, date : DATE, folder : DocumentationFolder, cost : REAL, limit : REAL)

    CALL SUPER(number, date, folder)

    SET treatmentCost TO cost
    SET medicalCoverageLimit TO limit

END CONSTRUCTOR


METHODS

    calculatePayout()

        IF treatmentCost <= medicalCoverageLimit THEN

            RETURN treatmentCost

        ELSE

            RETURN medicalCoverageLimit

        END IF

    END METHOD


END CLASS



CLASS PropertyClaim EXTENDS Claim

Attributes

    PRIVATE damageValue : REAL
    PRIVATE maintenancePercentage : REAL
    PRIVATE properlyMaintained : BOOLEAN


CONSTRUCTOR(number : TEXT, date : DATE, folder : DocumentationFolder, damage : REAL, percentage : REAL, maintained : BOOLEAN)

    CALL SUPER(number, date, folder)

    SET damageValue TO damage
    SET maintenancePercentage TO percentage
    SET properlyMaintained TO maintained

END CONSTRUCTOR


METHODS

    calculatePayout()

        IF properlyMaintained = TRUE THEN

            RETURN damageValue

        ELSE

            RETURN damageValue - (damageValue * maintenancePercentage)

        END IF

    END METHOD


END CLASS

CLASS HourlyInvoice EXTENDS Invoice

Attributes

    PRIVATE hourlyRate : REAL
    PRIVATE hourEntries : LIST<HourEntry>


CONSTRUCTOR(number : TEXT, date : DATE, client : TEXT, terms : PaymentTerms, rate : REAL)

    CALL SUPER(number, date, client, terms)

    SET hourlyRate TO rate
    SET hourEntries TO EMPTY LIST

END CONSTRUCTOR


METHODS

    addHourEntry(entry : HourEntry)

        ADD entry TO hourEntries

    END METHOD


    calculateTotalAmount()

        SET total TO 0


        FOR EACH entry IN hourEntries

            IF entry.isApproved() = TRUE THEN

                SET total TO total + (entry.getHours() * hourlyRate)

            END IF

        END FOR


        RETURN total

    END METHOD


END CLASS



CLASS RetainerInvoice EXTENDS FixedPriceInvoice

Attributes

    PRIVATE includedHours : REAL
    PRIVATE extraHoursWorked : REAL
    PRIVATE extraHourlyRate : REAL


CONSTRUCTOR(number : TEXT, date : DATE, client : TEXT, terms : PaymentTerms, fee : REAL, included : REAL, extra : REAL, rate : REAL)

    CALL SUPER(number, date, client, terms, fee)

    SET includedHours TO included
    SET extraHoursWorked TO extra
    SET extraHourlyRate TO rate

END CONSTRUCTOR


METHODS

    calculateTotalAmount()

        SET total TO SUPER.calculateTotalAmount()

        IF extraHoursWorked > includedHours THEN

            SET extraCharge TO (extraHoursWorked - includedHours) * extraHourlyRate

            SET total TO total + extraCharge

        END IF


        RETURN total

    END METHOD


END CLASS

CLASS WebServer EXTENDS Server

    PRIVATE activeConnections : NUMBER
    PRIVATE maxConnections : NUMBER

    CALL SUPER(serial, time, hardware)

    SET activeConnections TO active
    SET maxConnections TO max

END CONSTRUCTOR


METHODS

    calculateLoadPercentage()

        RETURN (activeConnections / maxConnections) * 100

CLASS DatabaseServer EXTENDS Server

Attributes

    PRIVATE currentQueriesPerSecond : REAL
    PRIVATE maxQueriesPerSecond : REAL
    PRIVATE acceptingConnections : BOOLEAN


CONSTRUCTOR(serial : TEXT, time : REAL, hardware : HardwareSpecification, current : REAL, max : REAL)

    CALL SUPER(serial, time, hardware)

    SET currentQueriesPerSecond TO current
    SET maxQueriesPerSecond TO max
    SET acceptingConnections TO TRUE

END CONSTRUCTOR


METHODS

    calculateLoadPercentage()

        SET load TO (currentQueriesPerSecond / maxQueriesPerSecond) * 100


        IF load >= 100 THEN

            SET acceptingConnections TO FALSE

        END IF


        RETURN load

    END METHOD


    canAcceptConnections()

        RETURN acceptingConnections

    END METHOD


END CLASS



CLASS CacheServer EXTENDS DatabaseServer

Attributes

    PRIVATE cacheHitRatio : REAL


CONSTRUCTOR(serial : TEXT, time : REAL, hardware : HardwareSpecification, current : REAL, max : REAL, hitRatio : REAL)

    CALL SUPER(serial, time, hardware, current, max)

    SET cacheHitRatio TO hitRatio

END CONSTRUCTOR


METHODS

    calculateLoadPercentage()

        SET databaseLoad TO SUPER.calculateLoadPercentage()

        SET adjustedLoad TO databaseLoad * (1 - cacheHitRatio)


        RETURN adjustedLoad

    END METHOD


END CLASS

CLASS Rack

Attributes

    PRIVATE rackID : TEXT
    PRIVATE servers : LIST<Server>


CONSTRUCTOR(id : TEXT)

    SET THIS.rackID TO id
    SET servers TO EMPTY LIST

END CONSTRUCTOR


METHODS

    addServer(newServer : Server)

        ADD newServer TO servers

    END METHOD


    calculateAverageLoad()

        SET totalLoad TO 0


        FOR EACH server IN servers

        END FOR
