CLASS Claim

Attributes

    PRIVATE claimNumber : TEXT
    PRIVATE dateFiled : DATE
    PRIVATE status : TEXT
    PRIVATE documentationFolder : DocumentationFolder


CONSTRUCTOR(number : TEXT, date : DATE, folder : DocumentationFolder)

    SET THIS.claimNumber TO number
    SET THIS.dateFiled TO date
    SET THIS.status TO "Submitted"
    SET THIS.documentationFolder TO folder

END CONSTRUCTOR


METHODS

    updateStatus(newStatus : TEXT)

        SET status TO newStatus

    END METHOD


    getStatus()

        RETURN status

    END METHOD


    getClaimNumber()

        RETURN claimNumber

    END METHOD


    ABSTRACT METHOD calculatePayout()


END CLASS



CLASS VehicleClaim EXTENDS Claim

Attributes

    PRIVATE repairEstimate : REAL
    PRIVATE excessAmount : REAL


CONSTRUCTOR(number : TEXT, date : DATE, folder : DocumentationFolder, repair : REAL, excess : REAL)

    CALL SUPER(number, date, folder)

    SET repairEstimate TO repair
    SET excessAmount TO excess

END CONSTRUCTOR


METHODS

    calculatePayout()

        SET payout TO repairEstimate - excessAmount

        IF payout < 0 THEN
            RETURN 0
        ELSE
            RETURN payout
        END IF

    END METHOD


END CLASS



CLASS MedicalClaim EXTENDS Claim

Attributes

    PRIVATE treatmentCost : REAL
    PRIVATE medicalCoverageLimit : REAL


CONSTRUCTOR(number : TEXT, date : DATE, folder : DocumentationFolder, cost : REAL, limit : REAL)

    CALL SUPER(number, date, folder)

    SET treatmentCost TO cost
    SET medicalCoverageLimit TO limit

END CONSTRUCTOR


METHODS

    calculatePayout()

        IF treatmentCost <= medicalCoverageLimit THEN

            RETURN treatmentCost

        ELSE

            RETURN medicalCoverageLimit

        END IF

    END METHOD


END CLASS



CLASS PropertyClaim EXTENDS Claim

Attributes

    PRIVATE damageValue : REAL
    PRIVATE maintenancePercentage : REAL
    PRIVATE properlyMaintained : BOOLEAN


CONSTRUCTOR(number : TEXT, date : DATE, folder : DocumentationFolder, damage : REAL, percentage : REAL, maintained : BOOLEAN)

    CALL SUPER(number, date, folder)

    SET damageValue TO damage
    SET maintenancePercentage TO percentage
    SET properlyMaintained TO maintained

END CONSTRUCTOR


METHODS

    calculatePayout()

        IF properlyMaintained = TRUE THEN

            RETURN damageValue

        ELSE

            RETURN damageValue - (damageValue * maintenancePercentage)

        END IF

    END METHOD


END CLASS

CLASS RollerCoaster INHERITS Ride

Attributes

    PRIVATE carCapacity : NUMBER
    PRIVATE rideCycleTime : REAL



CONSTRUCTOR(n : TEXT, l : TEXT, log : SafetyInspectionLog, q : NUMBER, capacity : NUMBER, cycle : REAL)

    SUPER(n, l, log, q)

    SET carCapacity TO capacity
    SET rideCycleTime TO cycle

    OVERRIDE calculateWaitTime()

        SET numberOfCars TO getQueueSize() / carCapacity

CLASS HourlyInvoice INHERITS Invoice

Attributes

    PRIVATE hourlyRate : REAL
    PRIVATE hourEntries : LIST OF HourEntry



CONSTRUCTOR(number : TEXT, date : TEXT, client : TEXT, terms : PaymentTerms, rate : REAL)

    SUPER(number, date, client, terms)

    SET hourlyRate TO rate
    SET hourEntries TO EMPTY LIST

END CONSTRUCTOR



METHODS


    addHourEntry(entry : HourEntry)

        ADD entry TO hourEntries

    END METHOD



    OVERRIDE calculateTotalAmount()

        SET total TO 0


        REPEAT FROM i = 1 TO LENGTH(hourEntries)

            IF hourEntries[i].isApproved() = TRUE THEN

                SET total TO total + (hourEntries[i].getHours() * hourlyRate)

            END IF

        END REPEAT


        RETURN total

    END METHOD


END CLASS





CLASS RetainerInvoice INHERITS FixedPriceInvoice

Attributes

    PRIVATE includedHours : REAL
    PRIVATE extraHoursWorked : REAL
    PRIVATE extraHourlyRate : REAL



CONSTRUCTOR(number : TEXT, date : TEXT, client : TEXT, terms : PaymentTerms, fee : REAL, included : REAL, totalHours : REAL, rate : REAL)

    SUPER(number, date, client, terms, fee)

    SET includedHours TO included
    SET extraHourlyRate TO rate


    IF totalHours > includedHours THEN

        SET extraHoursWorked TO totalHours - includedHours

    ELSE

        SET extraHoursWorked TO 0

    END IF


END CONSTRUCTOR



METHODS


    OVERRIDE calculateTotalAmount()

        SET total TO SUPER.calculateTotalAmount()


        IF extraHoursWorked > 0 THEN

            SET extraCharge TO extraHoursWorked * extraHourlyRate

            SET total TO total + extraCharge

        END IF


        RETURN total

    END METHOD


END CLASS

CLASS WebServer INHERITS Server

Attributes

    PRIVATE activeConnections : NUMBER
    PRIVATE maxConnections : NUMBER



CONSTRUCTOR(serial : TEXT, time : REAL, hardware : HardwareSpecification, active : NUMBER, max : NUMBER)

    SUPER(serial, time, hardware)

    SET activeConnections TO active
    SET maxConnections TO max

END CONSTRUCTOR



METHODS

    OVERRIDE calculateLoadPercentage()

        RETURN (activeConnections / maxConnections) * 100

    END METHOD


END CLASS





CLASS DatabaseServer INHERITS Server

Attributes

    PRIVATE currentQueriesPerSecond : REAL
    PRIVATE maxQueriesPerSecond : REAL
    PRIVATE acceptingConnections : BOOLEAN



CONSTRUCTOR(serial : TEXT, time : REAL, hardware : HardwareSpecification, current : REAL, max : REAL)

    SUPER(serial, time, hardware)

    SET currentQueriesPerSecond TO current
    SET maxQueriesPerSecond TO max
    SET acceptingConnections TO TRUE

END CONSTRUCTOR



METHODS

    OVERRIDE calculateLoadPercentage()

        SET load TO (currentQueriesPerSecond / maxQueriesPerSecond) * 100


        IF load >= 100 THEN

            SET acceptingConnections TO FALSE

        END IF


        RETURN load

    END METHOD



    canAcceptConnections()

        RETURN acceptingConnections

    END METHOD


END CLASS





CLASS CacheServer INHERITS DatabaseServer

Attributes

    PRIVATE cacheHitRatio : REAL



CONSTRUCTOR(serial : TEXT, time : REAL, hardware : HardwareSpecification, current : REAL, max : REAL, hitRatio : REAL)

    SUPER(serial, time, hardware, current, max)

    SET cacheHitRatio TO hitRatio

END CONSTRUCTOR



METHODS

    OVERRIDE calculateLoadPercentage()

        SET databaseLoad TO SUPER.calculateLoadPercentage()

        SET adjustedLoad TO databaseLoad * (1 - cacheHitRatio)


        RETURN adjustedLoad

    END METHOD


END CLASS

CLASS Rack

Attributes

    PRIVATE rackID : TEXT
    PRIVATE servers : LIST OF Server



CONSTRUCTOR(id : TEXT)

    SET THIS.rackID TO id
    SET servers TO EMPTY LIST

END CONSTRUCTOR



METHODS

    addServer(newServer : Server)

        ADD newServer TO servers

    END METHOD



    calculateAverageLoad()

        SET totalLoad TO 0


        REPEAT FROM i = 1 TO LENGTH(servers)

            SET totalLoad TO totalLoad + servers[i].calculateLoadPercentage()

        END REPEAT


        RETURN totalLoad / LENGTH(servers)

    END METHOD


END CLASS
