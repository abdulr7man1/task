METHODS

    calculateCalories()

    END METHOD

CLASS RunningWorkout EXTENDS Workout

Attributes
    PRIVATE distance : REAL
    PRIVATE pace : REAL

CONSTRUCTOR(d : DATE, t : REAL, dis : REAL, p : REAL)
    CALL SUPER(d, t)
    SET distance TO dis
    SET pace TO p
END CONSTRUCTOR

METHODS

    calculateCalories()
        RETURN distance * pace * 10
    END METHOD

END CLASS



CLASS CyclingWorkout EXTENDS Workout

Attributes
    PRIVATE distance : REAL
    PRIVATE resistanceLevel : REAL

CONSTRUCTOR(d : DATE, t : REAL, dis : REAL, r : REAL)
    CALL SUPER(d, t)
    SET distance TO dis
    SET resistanceLevel TO r
END CONSTRUCTOR

METHODS

    calculateCalories()
        RETURN distance * resistanceLevel * 8
    END METHOD

END CLASS



CLASS WeightTrainingWorkout EXTENDS Workout

Attributes
    PRIVATE numberOfSets : NUMBER
    PRIVATE numberOfReps : NUMBER

CONSTRUCTOR(d : DATE, t : REAL, s : NUMBER, r : NUMBER)
    CALL SUPER(d, t)
    SET numberOfSets TO s
    SET numberOfReps TO r
END CONSTRUCTOR

METHODS

    calculateCalories()
        RETURN numberOfSets * numberOfReps * 2
    END METHOD

END CLASS



CLASS FitnessLog

Attributes
    PRIVATE workouts : LIST<Workout>

CONSTRUCTOR()
    SET workouts TO EMPTY LIST
END CONSTRUCTOR

METHODS

    addWorkout(newWorkout : Workout)
        ADD newWorkout TO workouts
    END METHOD

    calculateTotalCalories()

        SET total TO 0

        FOR EACH workout IN workouts
            SET total TO total + workout.calculateCalories()
        END FOR

        RETURN total

    END METHOD

END CLASS

CLASS Mage EXTENDS Character

Attributes
    PRIVATE manaPool : NUMBER

CONSTRUCTOR(n : TEXT, h : NUMBER, i : Inventory, m : NUMBER)
    CALL SUPER(n, h, i)
    SET manaPool TO m
END CONSTRUCTOR

METHODS

    attack()

        IF manaPool > 0 THEN
            SET manaPool TO manaPool - 1
            RETURN "Mage casts a magic spell"
        ELSE
            RETURN "Not enough mana"
        END IF

    END METHOD

END CLASS

TASK306
CLASS Bill

Attributes
    PRIVATE billingPeriod : TEXT
    PRIVATE dueDate : DATE
    PRIVATE meterReading : MeterReading

CONSTRUCTOR(p : TEXT, d : DATE, m : MeterReading)
    SET billingPeriod TO p
    SET dueDate TO d
    SET meterReading TO m
END CONSTRUCTOR

METHODS

    calculateAmountOwed()
        RETURN 0
    END METHOD

END CLASS



CLASS ElectricityBill EXTENDS Bill

Attributes
    PRIVATE ratePerUnit : REAL

CONSTRUCTOR(p : TEXT, d : DATE, m : MeterReading, r : REAL)
    CALL SUPER(p, d, m)
    SET ratePerUnit TO r
END CONSTRUCTOR

METHODS

    calculateAmountOwed()

        SET unitsUsed TO meterReading.getConsumption()
        RETURN unitsUsed * ratePerUnit

    END METHOD

END CLASS



CLASS WaterBill EXTENDS Bill

Attributes
    PRIVATE ratePerLiter : REAL
    PRIVATE standingCharge : REAL

CONSTRUCTOR(p : TEXT, d : DATE, m : MeterReading, r : REAL, s : REAL)
    CALL SUPER(p, d, m)
    SET ratePerLiter TO r
    SET standingCharge TO s
END CONSTRUCTOR

METHODS

    calculateAmountOwed()

        SET litersUsed TO meterReading.getConsumption()
        RETURN (litersUsed * ratePerLiter) + standingCharge

    END METHOD

END CLASS



CLASS GasBill EXTENDS Bill

Attributes
    PRIVATE ratePerUnit : REAL
    PRIVATE winterSurcharge : REAL

CONSTRUCTOR(p : TEXT, d : DATE, m : MeterReading, r : REAL, s : REAL)
    CALL SUPER(p, d, m)
    SET ratePerUnit TO r
    SET winterSurcharge TO s
END CONSTRUCTOR

METHODS

    calculateAmountOwed()

        SET unitsUsed TO meterReading.getConsumption()
        RETURN (unitsUsed * ratePerUnit) + winterSurcharge

    END METHOD

END CLASS



CLASS MeterReading

Attributes
    PRIVATE startReading : REAL
    PRIVATE endReading : REAL

CONSTRUCTOR(s : REAL, e : REAL)
    SET startReading TO s
    SET endReading TO e
END CONSTRUCTOR

METHODS

    getConsumption()
        RETURN endReading - startReading
    END METHOD

END CLASS



CLASS CustomerAccount

Attributes
    PRIVATE bills : LIST<Bill>

CONSTRUCTOR()
    SET bills TO EMPTY LIST
END CONSTRUCTOR

METHODS

    addBill(newBill : Bill)
        ADD newBill TO bills
    END METHOD

    calculateTotalAmount()

        SET total TO 0

        FOR EACH bill IN bills
            SET total TO total + bill.calculateAmountOwed()
        END FOR

        RETURN total

    END METHOD

END CLASS

CLASS StandardRide EXTENDS Ride

Attributes
    PRIVATE distance : REAL
    PRIVATE ratePerKilometer : REAL

CONSTRUCTOR(p : TEXT, d : TEXT, r : Route, dis : REAL, rate : REAL)
    CALL SUPER(p, d, r)
    SET distance TO dis
    SET ratePerKilometer TO rate
END CONSTRUCTOR

METHODS

    calculateFare()
        RETURN distance * ratePerKilometer
    END METHOD

END CLASS



CLASS SharedRide EXTENDS StandardRide

Attributes
    PRIVATE numberOfPassengers : NUMBER

CONSTRUCTOR(p : TEXT, d : TEXT, r : Route, dis : REAL, rate : REAL, n : NUMBER)
    CALL SUPER(p, d, r, dis, rate)
    SET numberOfPassengers TO n
END CONSTRUCTOR

METHODS

    calculateFare()
        RETURN CALL SUPER.calculateFare() / numberOfPassengers
    END METHOD

END CLASS



CLASS PremiumRide EXTENDS Ride

Attributes
    PRIVATE distance : REAL
    PRIVATE ratePerKilometer : REAL
    PRIVATE comfortSurcharge : REAL

CONSTRUCTOR(p : TEXT, d : TEXT, r : Route, dis : REAL, rate : REAL, s : REAL)
    CALL SUPER(p, d, r)
    SET distance TO dis
    SET ratePerKilometer TO rate
    SET comfortSurcharge TO s
END CONSTRUCTOR

METHODS

    calculateFare()
        RETURN (distance * ratePerKilometer) + comfortSurcharge
    END METHOD

END CLASS
